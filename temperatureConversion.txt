// Convert a Celsius temperature in Fahrenheit

algorithm temperatureConversion() : void {

    variables temp_F, temp_C : Real;

    print("Type a temperature in 째C to convert in 째F :");
    temp_C <- readReal();
    temp_F <- temp_C * 1.8 + 32.;
    print(RealToString(temp_C)+"째C equals to "+RealToString(temp_F)+"째F.");

}
